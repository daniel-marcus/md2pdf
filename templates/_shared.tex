\tcbset{
  customunderline/.style={
    on line,
    box align=base,
    colframe=white,
    colback=white,
    boxrule=0pt,
    sharp corners,
    top=0pt,
    bottom=0pt,
    left=0pt,
    right=0pt,
    boxsep=0pt,
    enhanced,
    underlay={
      \draw[linkcolor, line width=6pt]
        ([xshift=4pt,yshift=-1pt]frame.south west) --
        ([xshift=4pt,yshift=-1pt]frame.south east);
    },
  }
}

\newcommand{\customhref}[2]{%
  \href{#1}{%
    \tcbox[customunderline]{%
      \raisebox{0pt}[0pt][0pt]{#2}%
    }%
  }%
}
